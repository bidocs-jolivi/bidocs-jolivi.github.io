      select A.SubscriberKey,
   EmailAddress,
   joined_date,
   subscription_id,
   payment_date,
   product_name,
   status,
   date_canceled,
   utm_campaign,
   payment_method,
   frontend,
   payment_installments payment_installme,
   duration,
   plan_id,
   due_date,
   price,
   name,
   A.chosen_product,
   xpromo,
   parent_id,
   cycles,
   cycle_id
from ent.DE_JOLIVI_PAID A
inner join (select SubscriberKey, chosen_product from ent.DE_JOLIVI_PAID where product_name = 'Golden Six - Produto') B
on A.SubscriberKey = B.SubscriberKey
and A.plan_id = B.chosen_product
left join (select SubscriberKey, chosen_product from ent.DE_JOLIVI_PAID where product_name = 'Clube Jolivi') C
on A.SubscriberKey = C.SubscriberKey
where C.SubscriberKey is null
and A.payment_method in ('smartbill_subscription_creditcardvista','vindi_creditcard_cashh','vindi_creditcard_split','smartbill_subscription_creditcard')
and cast(due_date as date) between dateadd(d,+11,getdate()) and dateadd(d,+12,getdate())
