    • Query_SUBSCRIBER_MICROSOFT

      select distinct subscriberkey
from Novo_engajamento_Geralv1
where (subscriberkey like '%hotmail.com%' or subscriberkey like '%msn.com%'
or subscriberkey like '%outlook.com%' or subscriberkey like '%live.com%' or subscriberkey like '%windowslive.com%')
and matriz in ('Matriz 1','Matriz 2','Matriz 3','Matriz 9','Matriz 10','Matriz 11','Onboarding')

      Target Data Extension: EMAIL_MICROSOFT (Overwrite)

    • Microsoft_Open7d_SClick

      select NG.subscriberkey,
   data_open,
   data_click,
   data_de_entrada,
   data_ultimo_envio,
   matriz,
   pago
from Novo_engajamento_Geralv1 NG
inner join ent.EMAIL_MICROSOFT EM
on NG.subscriberkey = EM.SubscriberKey
where cast(data_open as date) between DATEADD(day, - 20, GETDATE()) and GETDATE()
and cast(data_click as date) is null

      Target Data Extension: Microsoft_Open7D_SClick (Overwrite)
