      SELECT
SubscriberKey,
name,
CASE WHEN parent_id = 'palavra-lair-anual-livro' THEN 'Combo Palavra e Livro'
     WHEN parent_id = '303237099' THEN 'Diabetes Zero Essencial'
     WHEN parent_id = '203236099' THEN 'Diabetes Zero'
     WHEN parent_id = '200632006' then 'Homem de Ferro - Essential'
     WHEN parent_id = '202132099' THEN 'Relançamento CAI'
     WHEN parent_id = '200631099' THEN 'HF Dossie BB'
     WHEN PLAN_ID = '102200012' and coalesce(parent_id,'N/A') <> '200900006' THEN 'Dossie Anual'
     WHEN parent_id = '102208099' THEN 'Dossie Anual'
     WHEN parent_id = '102201099' and plan_id= '102200099' THEN 'Dossie Vitalicio + Expresso'
     WHEN PLAN_ID = '100500012' and coalesce(parent_id,'N/A') not in ('101600012', '200900006')  THEN 'Conexao Cerebro Anual'
     WHEN PLAN_ID = '103300012' and coalesce(parent_id,'N/A') not in ('101600012', '103300030', '200900006')  THEN 'Remedio Natural Anual' 
     WHEN parent_id = '100415099' then 'Palavra Lair Anual'
     WHEN (PLAN_ID = '100400012' and coalesce(parent_id,'N/A') not in  ('palavra-lair-anual-livro', '200900006','2021114099','101426099','100434012')) THEN 'Palavra Lair Anual'
     WHEN parent_id = '100409099' THEN 'Palavra Lair Anual'
     WHEN parent_id = '100406099' THEN 'Palavra Lair Anual'
     WHEN parent_id='book-offer-emagreca-comendo'  THEN 'Palavra do Lair Mensal'
     WHEN parent_id='200900006' THEN 'Golden Six Combo'
     WHEN parent_id ='200807099' THEN 'Golden Six'
     WHEN PLAN_ID = '200600006' AND coalesce(parent_id,'N/A') <> '202700099' THEN 'Homem de Ferro' 
     WHEN PLAN_ID = '201000006' AND coalesce(parent_id,'N/A') <> '202700099' THEN 'Vida Sem Dor'
     WHEN PLAN_ID = '201300099' AND coalesce(parent_id,'N/A') <> '202700099' THEN 'SOS Artrite'
     WHEN PLAN_ID = '201505099' AND coalesce(parent_id,'N/A') <> '202700099' THEN '30 Anos Mais Jovem'
     WHEN parent_id = '2021114099' then 'Coracao Anti-infarto'
     WHEN parent_id = '101600012' THEN 'Combo Remedio Conexao'
     WHEN parent_id = '101400012' or parent_id = '101426099' THEN 'Clube Jolivi'    
     WHEN parent_id = '201707099'  THEN 'Mais Visao'
     WHEN parent_id = '103300030' THEN 'FREE TRIAL REMEDIO'
     WHEN parent_id = '202515099' then 'Coracao Anti-infarto Essencial'
     WHEN parent_id = '101812012' THEN 'Super Saude Anual'
     WHEN parent_id = 'book-offer-remedio-natural' THEN 'Remédio Mensal'
     WHEN parent_id = '201913099' then 'Voce Mais Jovem'
     WHEN parent_id = '202519099' then 'Coluna 100%'
     WHEN parent_id = '103321099' then 'Remédio Natural Vitalício + Expresso + Grande Livro da Saúde Natural'
     WHEN parent_id = '103322012' then 'Remedio Natural Anual + Big Book'
     WHEN parent_id = '102225099' then 'Dossiê Vitalício + Expresso + Grande Livro da Saúde Natural'
     WHEN parent_id = '102224012' then 'Dossiê Anual + Big Book'
     WHEN parent_id = '100524012' then 'Conexão Anual + O Grande Livro'
     WHEN parent_id = '100525099' then 'Conexão Vitalício + Expresso + O Grande Livro'
     WHEN parent_id = '203028099' then 'SuperCerebro Sem Alzheimer'
     WHEN parent_id = '102927012' then 'Dossie Essencial Livro'
     WHEN parent_id in ('100420012','100434012') then 'Palavra + Livro CAI'
     WHEN parent_id = '203029099' then 'SuperCerebro Sem Alzheimer Essencial'
     WHEN parent_id = '100530012' then 'Conexão Anual + O Grande Livro'
     ELSE (convert(varchar,parent_id,101) + '-' + convert(varchar,PLAN_ID,101))
    END AS GAUNTLET,
    joined_date as Payment_date
FROM ent.DE_JOLIVI_PAID
WHERE
(PLAN_ID IN ('101426099','100525099','100524012','102224012','102225099','103322012','103321099','102200012','201300099','100500012','102200099','103300012','100400012','100700000','200800006','200600006','201000006','100400001', '103300001', '102200099', '201500099') 
or parent_id in('303237099','203236099','100434012','200632006','100530012','203029099','102927012','100420012','203028099','202700099','202519099','100415099','202515099','2021114099','200807099','101812012','201913099','palavra-lair-anual-livro','102201099', '201000006','201300099','201505099','101400012','101400012' ,'101600012','100406099','201707099','103300030','book-offer-remedio-natural','book-offer-emagreca-comendo','200900006' ,'200600006','102208099','100409099','200631099','202132099'))
AND upper(STATUS) in  ('COMPLETE', 'UPSELLING')
AND cast(joined_date as date) >= DATEADD(day,-3,convert(date,GETDATE()))
AND cast(joined_date as date)  < DATEADD(day,0,convert(date,GETDATE()))
AND (cycles < 2 or cycles is null)
