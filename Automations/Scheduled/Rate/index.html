    • Rate

      SELECT
r.SubscriberKey,
r.JobID,
r.Dt_Rate as DateRate,
r.Rate, 
j.EmailSubject as Subject,
j.EmailName,
s.EventDate as DateSent,
p.data_de_entrada as DateSubscribed
FROM ent.Rate r 
INNER JOIN _Job j
ON r.JobID = j.JobID
INNER JOIN _Sent s
ON r.JobID = s.JobID AND r.SubscriberKey = s.SubscriberKey
INNER JOIN Novo_engajamento_Geralv1 p
ON r.SubscriberKey = p.SubscriberKey
