      SELECT
T1.SUBSCRIBERKEY,
T1.DATA_DE_ENTRADA,
T1.XCODE,
CASE
when xcode = 'XJ-MEL-JOLI-SDA-X-X-Q2-X-X' then 'Quiz Alzheimer'
when xcode = 'XJ-MEL-JOLI-LPVMJ06-X-X-LP-X-X' then 'Hormonios'
when xcode = 'XJ-MI-JOLI-QUIZTESTO-X-X-QUIZ-X-X' then 'Quiz Testosterona'
when xcode = 'XJ-MEL-FB-HF02-SUBHL-L1-X-PP-2-X' then 'Homem de Ferro MKT'
when xcode = 'XJ-MEL-FB-HF02-SUBHL-L1-X-PP-1-X' then 'Homem de Ferro MKT'
when xcode = 'XJ-MEL-FB-HF02-SUBHL-L1-X-PP-3-X' then 'Homem de Ferro MKT'
when xcode = 'XJ-MEL-FB-HF02-SUBHL-L1-X-PP-4-X' then 'Homem de Ferro MKT'
when xcode = 'XJ-MEL-FB-HF02-SUBHL-L1-X-PP-5-X' then 'Homem de Ferro MKT'
when xcode = 'XJ-MEL-OB-LPHF01-X-X-PP-X-X' then 'Homem de Ferro MKT'
when xcode like '%LPUZC01%' then 'Dr Uronal'
when xcode like '%EMP-SSA%' then 'Super Saude Anual'
when xcode like '%XJ-MEL-FB-PTHF-SUBHL-L1-PP-1-X%' then 'Prostata'
when xcode like '%XJ-MEL-FB-PTHF-SUBHL-L1-PP-2-X%' then 'Prostata'
when xcode like '%XJ-MEL-OB-PTHF-X-X-PP-X-X%' then 'Prostata'
when xcode like '%VMJ%' then 'Voce mais jovem'
ELSE 'DEFAULT' END AS GAUNTLET
FROM CLASSIFICACAO_JOLIVI T1
LEFT JOIN ENT.JOLIVI_PLANOS T2
ON T1.SUBSCRIBERKEY = T2.SUBSCRIBERKEY
WHERE
T2.SUBSCRIBERKEY IS NULL
AND data_de_entrada >= DATEADD(day,-2,convert(date,GETDATE())) 
AND data_de_entrada  < DATEADD(day,0,convert(date,GETDATE())) 
AND T1.SUBSCRIBERKEY NOT IN (SELECT SUBSCRIBERKEY FROM ENT.DE_G_HOTLIST WHERE HL_CODE = 'JHL_HF')
