    • Query_Open_Email

      select distinct 
j.JourneyName,
j.VersionNumber,
ja.ActivityName as 'EmailName',
s.EventDate as 'SendTime',
s.SubscriberKey,
s.SubscriberID as 'ContactID'    ,
jo.JobID,
s.ListID,
s.BatchID,
case when O.SubscriberID is not null then 'True' else 'False' end as UniqueOpens,
'True' as UniqueClicks,
jo.EmailName as 'EmailNameJob',
jo.EmailSubject 
from [_Sent] s
left join [_job] jo
on s.JobID = jo.JobID
left join [_JourneyActivity] ja 
on s.TriggererSendDefinitionObjectID = ja.JourneyActivityObjectID
and ja.ActivityType in  ('EMAIL','EMAILV2')
left join [_Journey] j
on ja.VersionID = j.VersionID
left join _open O
on s.SubscriberID = O.SubscriberID
and s.JobID = O.JobID
where cast(s.EventDate as date) between '2018-12-05' and '2018-12-05'
and  jo.EmailName like '%_VD_%'
